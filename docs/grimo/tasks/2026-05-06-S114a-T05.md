## Key Design Notes

### grants.ts
```typescript
export interface SkillGrant {
  id: string
  principalType: 'user' | 'group' | 'company' | 'public'
  principalId: string
  role: 'OWNER' | 'VIEWER'
  grantedBy: string
  grantedAt: string
}

export interface CreateGrantRequest {
  principalType: SkillGrant['principalType']
  principalId: string
  role: 'VIEWER'  // MVP: 只開放 VIEWER（owner 約束 one-owner）
}

export const fetchGrants = (skillId: string): Promise<SkillGrant[]> =>
  apiClient.get(`/api/v1/skills/${skillId}/grants`).then(r => r.data)

export const createGrant = (skillId: string, body: CreateGrantRequest): Promise<{grantId: string}> =>
  apiClient.post(`/api/v1/skills/${skillId}/grants`, body).then(r => r.data)

export const revokeGrant = (skillId: string, grantId: string): Promise<void> =>
  apiClient.delete(`/api/v1/skills/${skillId}/grants/${grantId}`)
```

### useGrants.ts (TanStack Query)
```typescript
export function useGrants(skillId: string) {
  return useQuery({ queryKey: ['grants', skillId], queryFn: () => fetchGrants(skillId) })
}

export function useCreateGrant(skillId: string) {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (body: CreateGrantRequest) => createGrant(skillId, body),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['grants', skillId] })
  })
}

export function useRevokeGrant(skillId: string) {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (grantId: string) => revokeGrant(skillId, grantId),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['grants', skillId] })
  })
}
```
