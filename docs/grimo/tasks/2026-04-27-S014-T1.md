### 7. MapJsonbConverterTest — POC 兼回歸測試

`backend/src/test/java/.../shared/persistence/MapJsonbConverterTest.java`：

```java
@DataJdbcTest
@Import(JdbcConfiguration.class)
@Testcontainers
class MapJsonbConverterTest {

    @Autowired NamedParameterJdbcTemplate jdbc;

    @BeforeEach
    void setup() {
        jdbc.getJdbcTemplate().execute("""
            CREATE TABLE IF NOT EXISTS jsonb_test (
                id VARCHAR(36) PRIMARY KEY,
                payload JSONB NOT NULL
            )
        """);
    }

    @Test
    @DisplayName("AC-2: Map<String,Object> nested round-trip — String/List/Map 型別保留")
    @Tag("AC-2")
    void roundTrip_preservesNestedTypes() throws Exception {
        Map<String, Object> input = Map.of(
                "name", "test-skill",
                "version", "1.0.0",
                "tags", List.of("docker", "k8s"),
                "metadata", Map.of("author", "sam", "license", "MIT")
        );

        // 透過 Converter 寫入（直接組 PGobject）
        var pgo = new JdbcConfiguration.MapToPGobjectConverter(objectMapper).convert(input);
        jdbc.update("INSERT INTO jsonb_test (id, payload) VALUES (:id, :p)",
                new MapSqlParameterSource()
                        .addValue("id", "t1")
                        .addValue("p", pgo));

        // 讀回
        var result = jdbc.queryForObject(
                "SELECT payload FROM jsonb_test WHERE id = :id",
                Map.of("id", "t1"),
                (rs, rowNum) -> new JdbcConfiguration.PGobjectToMapConverter(objectMapper)
                        .convert((PGobject) rs.getObject("payload")));

        assertThat(result).containsEntry("name", "test-skill");
        assertThat(result.get("tags")).isInstanceOf(List.class);
        assertThat((List<?>) result.get("tags")).containsExactly("docker", "k8s");
        assertThat(result.get("metadata")).isInstanceOf(Map.class);
        assertThat((Map<?, ?>) result.get("metadata"))
                .containsEntry("author", "sam")
                .containsEntry("license", "MIT");
    }
}
```
