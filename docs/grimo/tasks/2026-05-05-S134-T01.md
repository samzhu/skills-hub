        @DefaultValue("true") boolean enabled,
        @DefaultValue Login login) {

### 3. SecurityConfig.filterChain — 三分支

在既有 `if (props.security().oauth().enabled())` 內加最內層分支：

```java
if (props.security().oauth().enabled()) {
    http.authorizeHttpRequests(auth -> auth
            .requestMatchers("/api/v1/me", "/api/v1/me/**").authenticated()
            .requestMatchers("/api/v1/notifications", "/api/v1/notifications/**").authenticated()
            .requestMatchers("/api/v1/admin/**").authenticated()
            .requestMatchers("/api/v1/dev/**").authenticated()  // S134：debug endpoint 也守
            .anyRequest().permitAll())
        .oauth2ResourceServer(oauth2 -> oauth2
            .jwt(jwt -> jwt.jwtAuthenticationConverter(jwtAuthenticationConverter())));

    if (props.security().oauth().login().enabled()) {
        // S134：real-oauth profile path — 加上 OAuth2 Login (Client) chain
        http.oauth2Login(login -> login.defaultSuccessUrl("/", true));
    }
} else {
    // 既有 LAB 模式不變動
}
```

### 4. RealOAuthConfigTest — 新增 unit test

```java
// backend/src/test/java/io/github/samzhu/skillshub/shared/security/RealOAuthConfigTest.java
package io.github.samzhu.skillshub.shared.security;

import org.junit.jupiter.api.DisplayName;
import org.junit.jupiter.api.Test;
import static org.assertj.core.api.Assertions.assertThat;

import java.nio.file.Files;
import java.nio.file.Path;

/**
 * S134 AC-8：固化 issuer-uri 設計約束。
 *
 * <p>背景：IdP metadata `issuer` 欄位為 "https://auth-dev.omnihubs.cloud"（無 trailing slash）；
 * Spring Security {@code JwtDecoderProviderConfigurationUtils.validateIssuer()} 與
 * {@code JwtIssuerValidator} 皆走 {@code String.equals} 嚴格比對。配錯會在第一個 JWT decode 時拋
 * {@code IllegalStateException}。本 test 讀 application-real-oauth.yaml.example template 確保
 * issuer-uri 規範對齊（防止後續修改誤加 slash）。
 */
class RealOAuthConfigTest {

    @Test
    @DisplayName("AC-8: issuer-uri 必須與 IdP metadata `issuer` 欄位嚴格相等（無 trailing slash）")
    void issuerUriShouldNotHaveTrailingSlash() throws Exception {
        var template = Files.readString(Path.of("config/application-real-oauth.yaml.example"));
        // template 內每個出現 issuer-uri 的行都不應結尾為 "/"
        var issuerLines = template.lines()
                .filter(line -> line.trim().startsWith("issuer-uri:"))
                .toList();
        assertThat(issuerLines).isNotEmpty();
        for (var line : issuerLines) {
            var value = line.substring(line.indexOf(':') + 1).trim();
            // 移除可能的 inline comment
            if (value.contains("#")) {
                value = value.substring(0, value.indexOf('#')).trim();
            }
            assertThat(value)
                    .as("issuer-uri value '%s' must not have trailing slash (mismatches IdP metadata)", value)
                    .doesNotEndWith("/");
        }
    }
}
```
