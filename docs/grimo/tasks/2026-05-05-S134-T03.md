## Implementation Plan

### 1. AuthDebugPage 元件

```typescript
// frontend/src/pages/AuthDebugPage.tsx
import { useQuery } from '@tanstack/react-query';

const NOT_REAL_OAUTH = 'NOT_REAL_OAUTH_PROFILE';

async function fetchAuthDebug(): Promise<unknown> {
    const res = await fetch('/api/v1/dev/auth-debug');
    if (res.status === 404) {
        throw new Error(NOT_REAL_OAUTH);
    }
    if (!res.ok) {
        throw new Error(`HTTP ${res.status}`);
    }
    return res.json();
}

/**
 * S134：dev-only「我的認證」頁。
 * 顯示後端 dump 的 OAuth2 session / JWT claim shape，給開發者比對真實 IdP claim 與 mock 的差異。
 *
 * 啟用條件：後端走 real-oauth profile（SPRING_PROFILES_ACTIVE 含 real-oauth）；
 * 否則 fetch 收 404 → 顯示提示文字。
 */
export default function AuthDebugPage() {
    const { data, error, isLoading } = useQuery({
        queryKey: ['auth-debug'],
        queryFn: fetchAuthDebug,
        retry: false,
    });

    if (isLoading) {
        return <div className="p-6">載入中...</div>;
    }

    if (error) {
        if (error.message === NOT_REAL_OAUTH) {
            return (
                <div className="p-6 max-w-2xl">
                    <h1 className="text-2xl mb-4 font-semibold">我的認證</h1>
                    <p className="text-zinc-400">
                        目前未啟用真實 OAuth profile（需要 SPRING_PROFILES_ACTIVE 含 real-oauth）
                    </p>
                </div>
            );
        }
        return <div className="p-6 text-red-400">錯誤：{error.message}</div>;
    }

    return (
        <div className="p-6 max-w-4xl">
            <h1 className="text-2xl mb-4 font-semibold">我的認證</h1>
            <pre className="text-xs bg-zinc-900 p-4 rounded overflow-auto border border-zinc-800">
                {JSON.stringify(data, null, 2)}
            </pre>
        </div>
    );
}
```

### 3. AuthDebugPage 測試

```typescript
// frontend/src/pages/__tests__/AuthDebugPage.test.tsx
import { describe, it, expect, beforeEach, afterEach } from 'vitest';
import { render, screen } from '@testing-library/react';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { setupServer } from 'msw/node';
import { http, HttpResponse } from 'msw';
import AuthDebugPage from '../AuthDebugPage';

const server = setupServer();
beforeEach(() => server.listen({ onUnhandledRequest: 'error' }));
afterEach(() => server.resetHandlers());

function renderPage() {
    const qc = new QueryClient({ defaultOptions: { queries: { retry: false } } });
    return render(
        <QueryClientProvider client={qc}>
            <AuthDebugPage />
        </QueryClientProvider>
    );
}

describe('AuthDebugPage', () => {
    it('AC-7: 顯示 JSON dump from /api/v1/dev/auth-debug', async () => {
        server.use(http.get('/api/v1/dev/auth-debug', () => HttpResponse.json({
            principal_name: 'test-sub',
            authorities: ['SCOPE_openid'],
            id_token_claims: { sub: 'test-sub', iss: 'https://auth-dev.omnihubs.cloud' },
        })));

        renderPage();
        expect(await screen.findByText('我的認證')).toBeInTheDocument();
        const pre = await screen.findByText(/principal_name/);
        expect(pre.textContent).toContain('test-sub');
        expect(pre.textContent).toContain('https://auth-dev.omnihubs.cloud');
    });

    it('AC-7-fallback: 404 顯示「未啟用真實 OAuth profile」提示', async () => {
        server.use(http.get('/api/v1/dev/auth-debug', () => new HttpResponse(null, { status: 404 })));

        renderPage();
        expect(await screen.findByText(/未啟用真實 OAuth profile/)).toBeInTheDocument();
        expect(screen.queryByText(/principal_name/)).toBeNull();
    });
});
```
